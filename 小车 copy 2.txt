        const agvCar = document.getElementById('agvCar');

        // 路径点
        const pathPoints = [
            {x: 280, y: 250},
            {x: 280, y: 450},
            {x: 380, y: 450}
        ];
        let currentPointIndex = 0;
        let carX = 280;
        let carY = 250;
        let battery = 82;
        let runSeconds = 2*3600 + 21*60 +16;
        let totalMileage = 125.9;

        // 每秒更新数据
        setInterval(() => {
            // 移动到下一个点
            const target = pathPoints[currentPointIndex];
            if (target) {
                const dx = target.x - carX;
                const dy = target.y - carY;
                const dist = Math.sqrt(dx*dx + dy*dy);
                
                if (dist > 2) {
                    const moveSpeed = 1;
                    carX += (dx / dist) * moveSpeed;
                    carY += (dy / dist) * moveSpeed;
                    
                    agvCar.style.left = carX + 'px';
                    agvCar.style.top = carY + 'px';
                    
                    // 更新坐标
                    posX.textContent = ((carX - 100) * 0.1 + 30).toFixed(2);
                    posY.textContent = ((carY - 200) * 0.1 + 10).toFixed(2);
                    
                    // 更新里程
                    totalMileage += 0.001;
                    mileage.textContent = totalMileage.toFixed(1) + ' km';
                } else {
                    currentPointIndex = (currentPointIndex + 1) % pathPoints.length;
                }
            }

            // 模拟电池消耗
            battery -= 0.01;
            if (battery < 20) {
                batteryBar.classList.remove('bg-success');
                batteryBar.classList.add('bg-danger');
            } else if (battery < 50) {
                batteryBar.classList.remove('bg-success');
                batteryBar.classList.add('bg-warning');
            }
            batteryBar.style.width = battery + '%';
            batteryText.textContent = Math.round(battery) + '%';

            runSeconds++;

            const minutes = Math.floor((runSeconds % 3600) / 60);

        }, 1000);

            const pointMapZoomReset = document.getElementById('point-map-zoom-reset');
            
            // 点管理地图缩放函数
            function updatePointMapScale(newScale) {
                // 限制缩放范围 0.5 - 2.0
                pointMapScale = Math.max(0.5, Math.min(2.0, newScale));
                if (pointMapContent) {
                    pointMapContent.style.transform = `scale(${pointMapScale})`;
                }
            }
            
            // 点管理地图缩放事件绑定
            if (pointMapZoomIn) {
                pointMapZoomIn.addEventListener('click', () => updatePointMapScale(pointMapScale + 0.1));
            }
            if (pointMapZoomOut) {
                pointMapZoomOut.addEventListener('click', () => updatePointMapScale(pointMapScale - 0.1));
            }
            if (pointMapZoomReset) {
                pointMapZoomReset.addEventListener('click', () => updatePointMapScale(1));
            }
            
            // 鼠标滚轮缩放支持
            const pointMapContainer = document.getElementById('point-map-container');
            
            if (pointMapContainer) {
                pointMapContainer.addEventListener('wheel', (e) => {
                    e.preventDefault();
                    const delta = e.deltaY > 0 ? -0.1 : 0.1;
                    updatePointMapScale(pointMapScale + delta);
                });
            }
        }

            const sliderConfigs = [
                { slider: 'auto-charge-level', value: 'auto-charge-level-value', min: 0, max: 100 },
                { slider: 'empty-voltage', value: 'empty-voltage-value', min: 20, max: 30, step: 0.1 },

                { slider: 'ultrasonic-min', value: 'ultrasonic-min-value', min: 0, max: 5, step: 0.1 },
                { slider: 'charge-back-distance', value: 'charge-back-distance-value', min: 0, max: 5, step: 0.1 },
                { slider: 'charge-back-speed', value: 'charge-back-speed-value', min: 0, max: 2, step: 0.1 },
                { slider: 'nav-to-charge-distance', value: 'nav-to-charge-distance-value', min: 0, max: 5, step: 0.1 },
                { slider: 'charge-forward-distance', value: 'charge-forward-distance-value', min: 0, max: 5, step: 0.1 },
                { slider: 'charge-angle-deviation', value: 'charge-angle-deviation-value', min: 0, max: 1, step: 0.01 },
                { slider: 'path-safe-distance', value: 'path-safe-distance-value', min: 0, max: 5, step: 0.1 },
                { slider: 'path-speed', value: 'path-speed-value', min: 0, max: 2, step: 0.1 },
                { slider: 'no-nav-stop-time', value: 'no-nav-stop-time-value', min: 0, max: 60 },
                { slider: 'auto-nav-speed', value: 'auto-nav-speed-value', min: 0, max: 3, step: 0.1 }
            ];

            sliderConfigs.forEach(config => {
                const slider = document.getElementById(config.slider);
                const valueInput = document.getElementById(config.value);

                if (slider && valueInput) {
                    // 滑块变化时更新输入框
                    slider.addEventListener('input', function() {
                        valueInput.value = this.value;
                    });

                    // 输入框变化时更新滑块
                    valueInput.addEventListener('change', function() {
                        let val = parseFloat(this.value);
                        if (config.min !== undefined) val = Math.max(config.min, val);
                        if (config.max !== undefined) val = Math.min(config.max, val);
                        slider.value = val;
                        this.value = val;
                    });
                }
            });
        }
